#include "stdafx.h"
#include <iostream>
#include <cmath>
using namespace std;


int main()
{
	int mult; //simply our input value to be calcualted
	int factorial = 1; //Initial value for factorial
	bool cont = true; // Halts program when false

	while(cont){
		cout << "\nPlease type an integer to calculate factorial (type a negative to end) ";
	cin >> mult; //take input for mult

	for (int i = 1; i <= mult; i++) { 
		if (i == 1)
			factorial = 1; //base case, resets factorial to 1
		else
			factorial = factorial * i; //multiplies factorial by the next number
	}

if (mult < 0) { // if a non-positive number is entered, end program
	cont = false;
	return 0;
}
	cout.flush();
	cout << "\n" << mult << "! =  " << factorial << "\n"; //output to console
	system("pause");
	}
    return 0;
}
